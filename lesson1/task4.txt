let mode = "normal"; - правильное именование переменной
let my_valu3 = 102; - давать имя с разделителем между словами в виде нижнего подчерквания
                        можно, но лучше использовать верблюжий стиль - myValue3.
let 3my_value3 = "102"; - имя переменной не верно, цыфры в начале переменной запрещенны.
let __hello__ = "world"; - имя переменной не верно, рекомендуют использовать нижнее подчерквание
                            только в качестве префикса для именования внутренних методов и переменных объекта,
                            и только вида "_hello"
let $$$ = "money"; - имя переменной возможно начанать со знака $, но в данном случае имя не коректно, так как
                        не несет ни какой смысловой нагрузки, и тот кто читает код не сможет понять
                        для чего преднозначина эта переменная.
let !0_world = true; - имя переменной не верно, начанается не с латинских символов, и может привести к ошибке.
